% use: \circles{R}{r}{s} in body
% draws two circles of radii R, r with origin separation of s
% draws line between intersection points
\newcommand{\circles}[3]{               
  \begin{tikzpicture}
    % set inputs
    \def\rA{#1}
    \def\rB{#2}
    \def\sep{#3}

    % use them to define new variables
    \pgfmathsetmacro{\xA}{ (\rA*\rA - \rB*\rB + \dist*\dist) / (2*\dist) }
    \pgfmathsetmacro{\radicand}{\rA*\rA - \xA*\xA}

    \ifdim\radicand pt > 0pt
      \pgfmathsetmacro{\h}{sqrt(\radicand)}

      % intersection line
      \draw[dashed] (\xA, \h) -- (\xA, -\h);

      % circle origins
      %\draw[fill] (0, 0) circle (1pt);
      %\draw[fill] (\dist, 0) circle (1pt);

    \else % handle invalid inputs
      \node[red] at (\dist/2, \rA) {Circles do not intersect};
    \fi

    \draw (0, 0) circle (\rA);
    \draw (\dist, 0) circle (\rB);
  \end{tikzpicture}
}
